% === PART & CHAPTER STYLING ===
\usepackage{titlesec}

% secnumdepth = -1 means only parts (level -1) are numbered; chapters and
% below are not auto-numbered. Chapter titles in the source already include
% "Chapter N: " prefixes, so this keeps them as-is while giving us proper
% "Part I", "Part II" numbering automatically.
\setcounter{secnumdepth}{-1}

% Part title page: centred, sans-serif, with the "Part I" label above the
% title. Standard trade layout — full page on its own, blank verso behind.
% Note: in titlesec [display] mode, the `format` block applies to the label
% and the `before-code` block applies to the title; \sffamily must appear in
% both so neither line falls back to the serif main font.
\titleformat{\part}[display]
  {\centering\sffamily}
  {\Large\scshape Part \thepart}
  {1.5em}
  {\Huge\bfseries\sffamily}
\titlespacing*{\part}{0pt}{80pt}{40pt}

\titleformat{\chapter}[block]
  {\raggedright\sffamily\LARGE\bfseries}
  {}
  {0pt}
  {}
\titlespacing*{\chapter}{0pt}{40pt}{20pt}

\titleformat{\section}
  {\raggedright\sffamily\bfseries\Large}
  {}{0pt}{}

\titleformat{\subsection}
  {\raggedright\sffamily\bfseries\large}
  {}{0pt}{}
